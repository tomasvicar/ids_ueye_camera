358 inline std::shared_ptr<DeviceDescriptor> Interface::GetOrAddFoundDevice(PEAK_DEVICE_DESCRIPTOR_HANDLE foundDevice)
359 {
360  std::lock_guard<std::mutex> lock(m_devicesMutex);
361 
362  const auto deviceKey = QueryStringFromCInterfaceFunction([&](char* key, size_t* keySize) {
363  return PEAK_C_ABI_PREFIX PEAK_DeviceDescriptor_GetKey(foundDevice, key, keySize);
364  });
365  auto keyDevicePairIterator = m_devicesByKey.find(deviceKey);
366  if (keyDevicePairIterator == m_devicesByKey.end())
367  {
368  auto deviceDescriptor = std::static_pointer_cast<DeviceDescriptor>(
369  std::make_shared<ClassCreator<DeviceDescriptor>>(
370  foundDevice, std::weak_ptr<Interface>{ shared_from_this() }));
371  m_devices.emplace_back(deviceDescriptor);
372  m_devicesByKey.emplace(deviceKey, deviceDescriptor);
373  m_devicesKeyById.emplace(deviceDescriptor->ID(), deviceKey);
374 
375  return deviceDescriptor;
376  }
377 
378  return keyDevicePairIterator->second;
379 }

402 inline void Interface::InitializeUpdateMechanismIfNecessary()
403 {
404  std::call_once(m_updateMechanismInitializedFlag, [&] {
405  (void)RegisterDeviceFoundCallback([](const std::shared_ptr<DeviceDescriptor>&) {
406  // Registering an empty callback is enough since DeviceFoundCallbackCWrapper() handles everything else.
407  });
408  (void)RegisterDeviceLostCallback([&](const std::string& lostDeviceId) {
409  std::lock_guard<std::mutex> lock(m_devicesMutex);
410  const auto lostDeviceKey = m_devicesKeyById.at(lostDeviceId);
411  m_devices.erase(std::remove_if(std::begin(m_devices), std::end(m_devices),
412  [lostDeviceKey](const std::shared_ptr<DeviceDescriptor>& device) {
413  return device->Key() == lostDeviceKey;
414  }),
415  std::end(m_devices));
416  m_devicesByKey.erase(lostDeviceKey);
417  m_devicesKeyById.erase(lostDeviceId);
418  });
419  });
420 }

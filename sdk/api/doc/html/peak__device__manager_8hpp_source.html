828 inline void DeviceManager::Reset(ResetPolicy resetPolicy)
829 {
830  std::lock_guard<std::mutex> lockUpdate(m_updateMutex);
831 
832  if (resetPolicy == ResetPolicy::ErrorOnOpenDevices)
833  {
834  // check if any devices are running
835  CheckDevicesOpened();
836  }
837 
838  {
839  std::lock_guard<std::mutex> lock(m_systemsMutex);
840  std::lock_guard<std::mutex> lock2(m_interfacesMutex);
841  std::lock_guard<std::mutex> lock3(m_devicesMutex);
842 
843  for (const auto& lostDevice : m_devices)
844  {
845  TriggerDeviceLostCallbacks(lostDevice->Key());
846  }
847  m_devices.clear();
848 
849  for (const auto& lostInterface : m_interfaces)
850  {
851  TriggerInterfaceLostCallbacks(lostInterface->Key());
852  }
853  m_interfaces.clear();
854 
855  m_systems.clear();
856  m_producerLibraries.clear();
857  }
858 
859  {
860  std::lock_guard<std::mutex> lock(m_producerLibrariesToAddMutex);
861  m_producerLibrariesToAdd.clear();
862  }

865 inline void DeviceManager::CheckDevicesOpened() const
866 {
867  size_t openDevicesFound = 0;
868  std::string deviceDisplayNames;
869 
870  {
871  std::lock_guard<std::mutex> lock(m_openedDevicesMutex);
872 
873  for (const auto& openedDevice : m_openedDevices)
874  {
875  auto openDevice = openedDevice.device.lock();
876  if (openDevice)
877  {
878  ++openDevicesFound;
879  deviceDisplayNames += openedDevice.displayName;
880  deviceDisplayNames += " ";
881  }
882  }
883  }
884 
885  if (openDevicesFound > 0)
886  {
887  throw core::InternalErrorException(
888  "Can't reset the device manager while there are devices open. The following devices ("
889  + std::to_string(openDevicesFound) + ") are open: " + deviceDisplayNames);
890  }

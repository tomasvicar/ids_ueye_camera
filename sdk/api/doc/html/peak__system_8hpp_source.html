443 inline std::shared_ptr<InterfaceDescriptor> System::GetOrAddFoundInterface(
444  PEAK_INTERFACE_DESCRIPTOR_HANDLE foundInterface)
445 {
446  std::lock_guard<std::mutex> lock(m_interfacesMutex);
447 
448  const auto interfaceKey = QueryStringFromCInterfaceFunction([&](char* key, size_t* keySize) {
449  return PEAK_C_ABI_PREFIX PEAK_InterfaceDescriptor_GetKey(foundInterface, key, keySize);
450  });
451  auto keyInterfacePairIterator = m_interfacesByKey.find(interfaceKey);
452  if (keyInterfacePairIterator == m_interfacesByKey.end())
453  {
454  auto interfaceDescriptor = std::static_pointer_cast<InterfaceDescriptor>(
455  std::make_shared<ClassCreator<InterfaceDescriptor>>(
456  foundInterface, std::weak_ptr<System>{ shared_from_this() }));
457  m_interfaces.emplace_back(interfaceDescriptor);
458  m_interfacesByKey.emplace(interfaceKey, interfaceDescriptor);
459  m_interfacesKeyById.emplace(interfaceDescriptor->ID(), interfaceKey);
460 
461  return interfaceDescriptor;
462  }
463 
464  return keyInterfacePairIterator->second;
465 }

488 inline void System::InitializeUpdateMechanismIfNecessary()
489 {
490  std::call_once(m_updateMechanismInitializedFlag, [&] {
491  (void)RegisterInterfaceFoundCallback([](const std::shared_ptr<InterfaceDescriptor>&) {
492  // Registering an empty callback is enough since InterfaceFoundCallbackCWrapper() handles everything else.
493  });
494  (void)RegisterInterfaceLostCallback([&](const std::string& lostInterfaceId) {
495  std::lock_guard<std::mutex> lock(m_interfacesMutex);
496  const auto lostInterfaceKey = m_interfacesKeyById.at(lostInterfaceId);
497  m_interfaces.erase(std::remove_if(std::begin(m_interfaces), std::end(m_interfaces),
498  [lostInterfaceKey](const std::shared_ptr<InterfaceDescriptor>& interface_) {
499  return interface_->Key() == lostInterfaceKey;
500  }),
501  std::end(m_interfaces));
502  m_interfacesByKey.erase(lostInterfaceKey);
503  m_interfacesKeyById.erase(lostInterfaceId);
504  });
505  });
506 }

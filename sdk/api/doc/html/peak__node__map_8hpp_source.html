511 inline void NodeMap::Initialize() const
512 {
513  auto numNodes = QueryNumericFromCInterfaceFunction<size_t>([&](size_t* _numNodes) {
514  return PEAK_C_ABI_PREFIX PEAK_NodeMap_GetNumNodes(m_backendHandle, _numNodes);
515  });
516 
517  for (size_t x = 0; x < numNodes; ++x)
518  {
519  auto nodeHandle = QueryNumericFromCInterfaceFunction<PEAK_NODE_HANDLE>(
520  [&](PEAK_NODE_HANDLE* _nodeHandle) {
521  return PEAK_C_ABI_PREFIX PEAK_NodeMap_GetNode(m_backendHandle, x, _nodeHandle);
522  });
523 
524  auto node = NodeFromHandle(nodeHandle);
525 
526  m_nodes.emplace_back(node);
527  m_nodesByName.emplace(node->Name(), node);
528  }

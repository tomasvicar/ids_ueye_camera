286  });

292  [&](PEAK_DATA_STREAM_DESCRIPTOR_HANDLE* _dataStreamDescriptorHandle) {

294  m_backendHandle, x, _dataStreamDescriptorHandle);
295  });
296 
297  dataStreams.emplace_back(std::make_shared<ClassCreator<DataStreamDescriptor>>(
298  dataStreamDescriptorHandle, std::const_pointer_cast<Device>(shared_from_this())));
299  }
